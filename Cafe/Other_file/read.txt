docker run --rm --name pg-docker -e POSTGRES_PASSWORD=docker -d -p 5400:5432 -v vol_1:/var/lib/postgresql/data postgres

    run - комнада отвечающая за запуск контейнера
    --rm - автоматически удаляет контейнер если он уже существует
    --name pg-docker - задает имя "pg-docker" для контейра
    -e POSTGRES_PASSWORD=docker - ключ -e устанавливает переменную окружения, в нашем случае мы установили пароль для баззы данных
    этот ключ можно испольовать много раз, что бы задать различные перемеренные
    -d - указывает, что контейнер следует запускать как backgraound процесс и не блокировать командную строку
    -p 5400:5432  - указывает какой порт предоставить наружу, в нашем случае мы производим отображение внутреннего порта 5432,
    на котором работает postgresql сервер в контейнера, на порт 5400. Т.е. что бы соедениться с сервером нам надо будет подключаться к порту 5400.
    -v vol_1:/var/lib/postgresql/data - тут мы производим соеденение внутренней структуры каталогов в контейнере на внешний докер том. Что бы хранить данные между перезапусками контейнера.
    postgres - имя докер образа. Стоить заметить что в данном случае мы всегда будем брать последний образ. Если вам нужна определенная версия, то ее следует указат в имени.

docker run --rm -v vol_1:/volume -v e:/backup:/backup busybox sh -c 'cp -r /volume /backup'

    --rm - автоматически удаляет контейнер если он уже существует

    busybox - докер образ в котором вы можете найти много linux утилит
    sh -c 'cp -r /volume /backup' - выполняем команду копирования данных
